## 二、代码规范

### 2.1 命名
- 组件文件: PascalCase, 如 `ProductList.vue`
- 工具/方法: camelCase
- 常量: UPPER_SNAKE_CASE
- CSS 类名: kebab-case
- 避免单字母变量 (循环 i 除外)

### 2.2 组件规范
使用 `<script setup>` (Composition API):
```vue
<template>
  <div class="product-page">
    <el-table :data="data.records">
      <el-table-column prop="name" label="名称" />
    </el-table>
  </div>
</template>

<script setup>
import { ref, reactive, onMounted } from 'vue'

const data = ref({ records: [], total: 0 })

  try {
    data.value = (await productApi.page(query)).data
  } finally {
    loading.value = false
  }
}

onMounted(loadData)
</script>

<style scoped>
.product-page { padding: 16px; }
</style>
```

### 2.3 Props 校验
```vue
<script setup>
import { defineProps, withDefaults } from 'vue'
```

### 2.4 Emit
```vue
<script setup>
import { defineEmits } from 'vue'

  emit('change', { value: 1 })
```

### 2.5 不要使用 Options API
- 统一 `<script setup>`
- 避免 `data()`, `methods:`, `computed:` 等
- 提升代码可读性和类型推断

## 三、API 封装模式

### 3.1 Axios 实例 (utils/request.js)
```js
import axios from 'axios'
import { ElMessage, ElMessageBox } from 'element-plus'
import { useUserStore } from '@/store/user'
import router from '@/router'
import NProgress from 'nprogress'

const service = axios.create({
  baseURL: '/api',
  timeout: 30000
})

// 请求拦截: 加 token
service.interceptors.request.use(config => {
  NProgress.start()
  const user = useUserStore()
  if (user.token) config.headers['Authorization'] = user.token
  return config
})

// 响应拦截: 统一处理
service.interceptors.response.use(res => {
  NProgress.done()
  const data = res.data
  if (data.code === 200) return data
  if (data.code === 401) {
    ElMessageBox.confirm('登录已过期', '提示').then(() => {
      router.push('/login')
    })
    return Promise.reject(data)
  }
  ElMessage.error(data.msg || '操作失败')
  return Promise.reject(data)
}, err => {
  NProgress.done()
  ElMessage.error(err.message || '网络异常')
  return Promise.reject(err)
})

export default service
```

## 四、状态管理 (Pinia)

### 4.1 用户态 (store/user.js)
```js
import { defineStore } from 'pinia'
import { login, logout, me } from '@/api/auth'

export const useUserStore = defineStore('user', {
  state: () => ({
    token: localStorage.getItem('erp_token') || '',
    userInfo: null,
    permissions: [],
    menus: []
  }),
  actions: {
    async loginAction(data) {
      const r = await login(data)
      this.token = r.data.token
      this.userInfo = r.data
      this.permissions = r.data.permissions || []
      this.menus = r.data.menus || []
      localStorage.setItem('erp_token', this.token)
    },
    async logoutAction() {
      await logout()
      this.token = ''
      this.userInfo = null
      localStorage.removeItem('erp_token')
    },
    hasPerm(p) {
      if (this.userInfo?.userId === 1) return true  // 超管
      return (this.permissions || []).includes(p)
    }
  }
})
```

### 4.2 在组件中使用
```vue
<script setup>
import { useUserStore } from '@/store/user'
import { useRouter } from 'vue-router'

const user = useUserStore()
const router = useRouter()

async function onLogout() {
  await user.logoutAction()
  router.push('/login')
}
</script>
```

### 5.2 权限拦截
```js
router.beforeEach((to, from, next) => {
  const user = useUserStore()
  if (to.meta.public) return next()
  if (!user.token) return next('/login')
  if (to.meta.perm && !user.hasPerm(to.meta.perm)) {
    ElMessage.warning('无访问权限: ' + to.meta.perm)
    return next(false)
  }
  next()
})
```

## 九、请求/响应处理

### 9.1 异步加载数据
```js
// 简单
const r = await api.list()
data.value = r.data

// 复杂 (带错误处理)
async function loadData() {
  loading.value = true
  try {
    const r = await api.list()
    if (r.code === 200) {
      data.value = r.data
    }
  } catch (e) {
    console.error(e)
  } finally {
    loading.value = false
  }
}
```

### 9.2 防抖 (搜索)
```js
import { debounce } from 'lodash-es'  // 或自己实现

const onSearch = debounce(() => loadData(), 300)
```

### 9.3 防止重复提交
```js
const submitting = ref(false)
async function onSubmit() {
  if (submitting.value) return
  submitting.value = true
  try {
    await api.add(form)
  } finally {
    submitting.value = false
  }
}
```
